stuct Cli {
    pattern: String,
    path: String
}

fn main() {
    let pattern = std::env::args().nth(1).expect("no pattern given");
    let path = std::env::args().nth(2).expect("no path_given");

    let args = Cli {
        pattern,
        path
    };

    println!("pattern: {:?}, path: {:?}", args.pattern, args.path)
}
